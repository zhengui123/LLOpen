===== 创建 RoomSlot.cs =====
using UnityEngine;
using UnityEngine.UI;
using UnityEngine.EventSystems;
using DG.Tweening;

public class RoomSlot : MonoBehaviour, IPointerDownHandler, IDragHandler, IPointerUpHandler
{
    [SerializeField] private Image coverImage;
    [SerializeField] private Image fleshImage;
    private bool isOpened = false;
    private bool isDragging = false;
    private float dragStartY;

    public bool IsOpened => isOpened;
    public YieldGrade RoomGrade { get; private set; }
    public int RoomIndex { get; private set; }

    public System.Action<RoomSlot> OnOpened;

    public void Init(int index, Sprite cover, Sprite flesh, YieldGrade grade)
    {
        RoomIndex = index;
        RoomGrade = grade;
        coverImage.sprite = cover;
        fleshImage.sprite = flesh;
        coverImage.gameObject.SetActive(true);
        fleshImage.gameObject.SetActive(false);
        isOpened = false;
        isDragging = false;
    }

    public void OnPointerDown(PointerEventData e) { dragStartY = e.position.y; isDragging = true; }
    public void OnDrag(PointerEventData e) { /* 跟随手指微移 */ }
    public void OnPointerUp(PointerEventData e)
    {
        if (!isDragging) return;
        isDragging = false;
        float delta = e.position.y - dragStartY;
        if (Mathf.Abs(delta) > 50f) TryOpen();
    }

    public bool TryOpen()
    {
        if (isOpened) return false;
        isOpened = true;

        // 壳盖弹飞
        Vector2 flyDir = (Vector2)coverImage.rectTransform.anchoredPosition.normalized + Random.insideUnitCircle * 0.5f;
        coverImage.rectTransform.DOAnchorPos(
            coverImage.rectTransform.anchoredPosition + flyDir * 80f, 0.3f).SetEase(Ease.OutQuad);
        coverImage.DOFade(0f, 0.2f).OnComplete(() => coverImage.gameObject.SetActive(false));

        // 果肉弹出
        fleshImage.gameObject.SetActive(true);
        fleshImage.rectTransform.localScale = Vector3.zero;
        fleshImage.rectTransform.DOScale(1f, 0.3f).SetEase(Ease.OutBack);

        OnOpened?.Invoke(this);
        return true;
    }
}

===== 在 Unity 中创建 RoomSlot Prefab =====
1. 创建空 GameObject 命名为 RoomSlot，尺寸 128×128
2. 添加 Image → CoverImage（显示RC壳盖），Canvas Order = 1
3. 添加 Image → FleshImage（显示RF果肉），Canvas Order = 0，初始 disable
4. 挂载 RoomSlot.cs
5. 保存为 Prefab

# Step 4: 重写 DurianOpener（逐房揭示 + 中途卖）

```
请完全重写 Assets/_Project/Scripts/Component/DurianOpener.cs：

using UnityEngine;
using UnityEngine.UI;
using DG.Tweening;
using Cysharp.Threading.Tasks;
using VContainer;
using System.Collections.Generic;

public class DurianOpener : MonoBehaviour
{
    [Inject] private DurianSpriteConfig spriteConfig;
    [Inject] private ShopManager shopManager;
    [Inject] private StreakCounter streakCounter;

    [Header("完整榴莲")]
    [SerializeField] private Image wholeDurianImage;

    [Header("房位容器")]
    [SerializeField] private Transform roomSlotsParent;
    [SerializeField] private GameObject roomSlotPrefab;

    [Header("金枕5房坐标")]
    [SerializeField] private Vector2[] roomPositions = new Vector2[] {
        new(0, -80), new(-110, -30), new(110, -30), new(-90, 80), new(90, 80)
    };

    [Header("中途卖")]
    [SerializeField] private GameObject sellMidwayGroup;
    [SerializeField] private Text sellMidwayPriceText;
    [SerializeField] private CanvasGroup sellMidwayCanvas;

    [Header("评级")]
    [SerializeField] private Image ratingBadgeImage;
    [SerializeField] private CanvasGroup ratingCanvasGroup;

    private List<RoomSlot> roomSlots = new();
    private DurianData currentDurian;
    private int openedRoomCount = 0;
    private int lastEstimate = 0;
    private bool hasSoldMidway = false;
    private bool allOpened = false;
    private YieldGrade[] roomGrades = new YieldGrade[5];

    // ===== 公开方法 =====

    public void PrepareOpen(DurianData durian)
    {
        currentDurian = durian;
        hasSoldMidway = false;
        allOpened = false;
        openedRoomCount = 0;
        lastEstimate = 0;

        // 显示完整榴莲（调色后）
        wholeDurianImage.sprite = spriteConfig.GetUnopenedSprite(durian.variety);
        wholeDurianImage.color = spriteConfig.GetAppearanceColor(durian.appearance);
        wholeDurianImage.gameObject.SetActive(true);

        // 根据出肉率分配5个房的果肉等级
        AssignRoomGrades(durian.yieldRate);

        // 生成5个 RoomSlot
        ClearRoomSlots();
        var covers = spriteConfig.GetRcJz();
        for (int i = 0; i < 5; i++)
        {
            var go = Instantiate(roomSlotPrefab, roomSlotsParent);
            go.GetComponent<RectTransform>().anchoredPosition = roomPositions[i];
            var slot = go.GetComponent<RoomSlot>();
            slot.Init(i, covers[i], spriteConfig.GetRf(roomGrades[i]), roomGrades[i]);
            slot.OnOpened += OnRoomOpened;
            roomSlots.Add(slot);
        }

        sellMidwayGroup.SetActive(false);
        ratingCanvasGroup.alpha = 0f;
    }

    public void OnRoomOpened(RoomSlot slot)
    {
        if (hasSoldMidway || allOpened) return;
        openedRoomCount++;

        bool hasFlesh = slot.RoomGrade >= YieldGrade.High;
        streakCounter.OnRoomRevealed(hasFlesh);
        UpdateEstimate();

        // 第一个房开后出现[见好就收]
        if (openedRoomCount == 1)
        {
            sellMidwayGroup.SetActive(true);
            sellMidwayCanvas.DOFade(1f, 0.2f);
        }

        // 5房全开
        if (openedRoomCount >= 5)
        {
            allOpened = true;
            sellMidwayGroup.SetActive(false);
            OnAllRoomsOpened();
        }
    }

    public void OnSellMidway()
    {
        hasSoldMidway = true;
        sellMidwayGroup.SetActive(false);
        // 快速打开剩余房
        for (int i = openedRoomCount; i < 5; i++)
            roomSlots[i].TryOpen();
        EventBus.Publish(new DurianMidwaySoldEvent(currentDurian, lastEstimate));
    }

    public async UniTask OnSwipeComplete()
    {
        if (hasSoldMidway || allOpened) return;
        // 自动打开所有剩余房
        for (int i = openedRoomCount; i < 5; i++)
        {
            await UniTask.Delay(200);
            roomSlots[i].TryOpen();
        }
    }

    public void Reset()
    {
        ClearRoomSlots();
        wholeDurianImage.gameObject.SetActive(false);
        sellMidwayGroup.SetActive(false);
        ratingCanvasGroup.alpha = 0f;
        hasSoldMidway = false;
        allOpened = false;
    }

    // ===== 私有方法 =====

    private void AssignRoomGrades(float yieldRate)
    {
        int fullRooms = Mathf.RoundToInt(yieldRate * 5f);
        for (int i = 0; i < 5; i++)
            roomGrades[i] = i < fullRooms ? YieldGrade.Perfect : YieldGrade.Empty;
        // 随机打乱
        for (int i = 4; i > 0; i--)
        { int j = Random.Range(0, i + 1); var t = roomGrades[i]; roomGrades[i] = roomGrades[j]; roomGrades[j] = t; }
        // 将 Perfection 替换为实际等级（RF级别映射到果肉量）
        // 满房不一定100%，可能80%等，这里简化为Perfection对应RF-100
    }

    private void UpdateEstimate()
    {
        int fullCount = 0;
        for (int i = 0; i < openedRoomCount; i++)
            if (roomGrades[i] >= YieldGrade.High) fullCount++;

        float ratio = (float)fullCount / Mathf.Max(1, openedRoomCount);
        int remaining = 5 - openedRoomCount;
        float uncertainty = remaining / 5f * Random.Range(-0.3f, 0.3f);
        float estFull = fullCount + remaining * ratio * (1f + uncertainty);
        int estimate = Mathf.RoundToInt(estFull * 50f * (1f + shopManager.GetSellBonus()));

        sellMidwayPriceText.text = estimate + " 金币";
        sellMidwayPriceText.color = estimate >= lastEstimate ? Color.green : Color.red;
        sellMidwayPriceText.transform.DOPunchScale(Vector3.one * 0.1f, 0.15f);
        lastEstimate = estimate;
    }

    private async void OnAllRoomsOpened()
    {
        await UniTask.Delay(300);
        YieldGrade overall = EstimateOverallGrade();
        PlayerProgression.Instance.UnlockCollection(currentDurian.variety, overall, lastEstimate);
        PlayerProgression.Instance.TotalOpens++;
        PlayerProgression.Instance.Save();
        ShowRating(overall);
        if (overall == YieldGrade.Perfect || streakCounter.CurrentStreak >= 3)
            ShowShareButton();
    }

    private YieldGrade EstimateOverallGrade()
    {
        int full = 0; for (int i = 0; i < 5; i++) if (roomGrades[i] >= YieldGrade.High) full++;
        return full switch { 0 => YieldGrade.Empty, 1 => YieldGrade.Low, 2 => YieldGrade.Normal, 3 => YieldGrade.High, _ => YieldGrade.Perfect };
    }

    private async void ShowRating(YieldGrade grade)
    {
        ratingBadgeImage.sprite = spriteConfig.GetRatingIcon(grade);
        ratingCanvasGroup.DOFade(1f, 0.3f);
        ratingBadgeImage.rectTransform.localScale = Vector3.zero;
        await ratingBadgeImage.rectTransform.DOScale(1f, 0.4f).SetEase(Ease.OutBack).ToUniTask();
    }

    private void ShowShareButton() { /* 微信分享 */ }
    private void ClearRoomSlots() { foreach (var s in roomSlots) if(s) Destroy(s.gameObject); roomSlots.Clear(); }
}

using UnityEngine;
using System;

public class StreakCounter
{
    public int CurrentStreak { get; private set; } = 0;

    public void OnRoomRevealed(bool hasFlesh)
    {
        if (hasFlesh) CurrentStreak++;
        else CurrentStreak = 0;

        if (CurrentStreak > PlayerProgression.Instance.BestStreak)
            PlayerProgression.Instance.BestStreak = CurrentStreak;

        EventBus.Publish(new StreakUpdatedEvent(CurrentStreak));
    }

    public int GetFlameLevel() => CurrentStreak switch { < 3 => 0, < 5 => 1, < 7 => 2, < 10 => 3, _ => 4 };

    public void Reset() { CurrentStreak = 0; EventBus.Publish(new StreakUpdatedEvent(0)); }
}

// 在 GameLifetimeScope 注册为 Singleton：

    int idx = GetCollectionIndex(variety, grade);

    {
        CollectionUnlocked[idx] = true;

using UnityEngine;
using System;

public class DailyTarget
{
    public int TargetGold = 500;
    public int EarnedToday = 0;
    public bool IsCompleted = false;
    public bool IsClaimed = false;
    private const int REWARD = 50;

    public void CheckDailyReset()
    {
        string today = DateTime.Now.ToString("yyyy-MM-dd");
        if (PlayerPrefs.GetString("DailyTargetDate", "") != today)
        {
            EarnedToday = 0; IsCompleted = false; IsClaimed = false;
            PlayerPrefs.SetString("DailyTargetDate", today);
            PlayerPrefs.Save();
        }
    }

    public void AddGold(int amount)
    {
        if (IsClaimed) return;
        EarnedToday += amount;
        PlayerProgression.Instance.TotalGoldEarned += amount;
        if (!IsCompleted && EarnedToday >= TargetGold)
        {
            IsCompleted = true;
            EventBus.Publish(new DailyTargetCompletedEvent());
        }
    }

    public bool ClaimReward()
    {
        if (!IsCompleted || IsClaimed) return false;
        IsClaimed = true;
        PlayerData.Instance.Gold += REWARD;
        return true;
    }
}

public class ShopManager
{
    public int CurrentLevel { get; private set; } = 1;
    public int MaxLevel => 3;  // v1.2 是2，v1.5 改为3

    private int[] upgradeCosts = { 0, 500, 1500 };
    private float[] sellBonuses = { 0f, 0.20f, 0.35f };

    public bool CanUpgrade() => CurrentLevel < MaxLevel && PlayerData.Instance.Gold >= upgradeCosts[CurrentLevel];

    public void Upgrade()
    {
        if (!CanUpgrade()) return;
        PlayerData.Instance.Gold -= upgradeCosts[CurrentLevel];
        CurrentLevel++;
        EventBus.Publish(new ShopUpgradedEvent(CurrentLevel));
    }

    public float GetSellBonus() => sellBonuses[CurrentLevel - 1];
}
