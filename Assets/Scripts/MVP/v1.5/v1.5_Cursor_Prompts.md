## Step D2: 创建 RoomSlot Prefab

### Cursor 提示词
```
在 Unity 创建 RoomSlot Prefab（128×128）：

RoomSlot (RectTransform, 128×128)
  ├── CoverImage (Image, RC-JZ-P1~P5) — 初始显示
  └── FleshImage (Image, RF-00~100)    — 初始隐藏, Canvas Order: Cover-1

在 CoverImage 上挂 RoomSlot.cs：
public class RoomSlot : MonoBehaviour, IPointerDownHandler, IDragHandler, IPointerUpHandler {
    [SerializeField] private Image coverImage;
    [SerializeField] private Image fleshImage;
    private bool isOpened = false;

    public void Init(Sprite cover, Sprite flesh) {
        coverImage.sprite = cover; fleshImage.sprite = flesh;
        coverImage.gameObject.SetActive(true); fleshImage.gameObject.SetActive(false);
        isOpened = false;
    }

    public bool TryOpen() {
        if(isOpened) return false;
        isOpened = true;
        // 壳盖弹飞
        coverImage.rectTransform.DOAnchorPos(
            coverImage.rectTransform.anchoredPosition + Random.insideUnitCircle * 80f, 0.3f).SetEase(Ease.OutQuad);
        coverImage.DOFade(0f, 0.2f).OnComplete(() => coverImage.gameObject.SetActive(false));
        fleshImage.gameObject.SetActive(true);
        fleshImage.rectTransform.localScale = Vector3.zero;
        fleshImage.rectTransform.DOScale(1f, 0.3f).SetEase(Ease.OutBack);
        return true;
    }
}
```

## Step D3: 重写 DurianOpener（逐房版）

### Cursor 提示词
```
重写 DurianOpener.cs：

[Inject] private DurianSpriteConfig spriteConfig;
[Inject] private ShopManager shopManager;
[Inject] private StreakCounter streakCounter;

[SerializeField] private Image wholeDurianImage;     // D-01 完整榴莲
[SerializeField] private Transform roomSlotsParent;   // RoomSlot 父节点
[SerializeField] private GameObject roomSlotPrefab;

[Header("中途卖")]
[SerializeField] private GameObject sellMidwayGroup;
[SerializeField] private Text sellMidwayPriceText;
[SerializeField] private Button sellMidwayButton;
[SerializeField] private Button continueButton;

[Header("评级")]
[SerializeField] private Image ratingBadgeImage;
[SerializeField] private CanvasGroup ratingCanvasGroup;

private List<RoomSlot> roomSlots = new();
private DurianData currentDurian;
private int openedRoomCount = 0;
private bool hasSoldMidway = false;
private bool allOpened = false;

// 生成时已预先判定好每房的果肉等级
private YieldGrade[] roomGrades = new YieldGrade[5];

public void PrepareOpen(DurianData durian) {
    currentDurian = durian;
    hasSoldMidway = false; allOpened = false; openedRoomCount = 0;

    // 显示完整榴莲（调色）
    wholeDurianImage.sprite = spriteConfig.GetUnopenedSprite(durian.variety);
    wholeDurianImage.color = spriteConfig.GetAppearanceColor(durian.appearance);
    wholeDurianImage.gameObject.SetActive(true);

    // 生成5个RoomSlot
    ClearRoomSlots();
    var covers = spriteConfig.GetRcJz();

    // 每房的果肉等级：基于出肉率分配给5个房
    AssignRoomGrades(durian.yieldRate);

    for(int i=0; i<5; i++) {
        var slot = Instantiate(roomSlotPrefab, roomSlotsParent).GetComponent<RoomSlot>();
        slot.Init(covers[i], spriteConfig.GetRf(roomGrades[i]));
        slot.GetComponent<RectTransform>().anchoredPosition = roomConfig.roomPositions[i];
        roomSlots.Add(slot);
    }

    sellMidwayGroup.SetActive(false);
    ratingCanvasGroup.alpha = 0f;
}

// 将出肉率映射到5个房的果肉等级
private void AssignRoomGrades(float yieldRate) {
    int fullRooms = Mathf.RoundToInt(yieldRate * 5f);
    var grades = new YieldGrade[5];
    for(int i=0; i<5; i++) grades[i] = i < fullRooms ? YieldGrade.Perfect : YieldGrade.Empty;
    // 随机打乱避免前面的房总是满的
    for(int i=4; i>0; i--) { int j=Random.Range(0,i+1); var t=grades[i]; grades[i]=grades[j]; grades[j]=t; }
    roomGrades = grades;
}

public void OnRoomOpened(int roomIndex, bool hasFlesh) {
    if(hasSoldMidway || allOpened) return;
    openedRoomCount++;

    // 更新估价
    UpdateEstimate();

    // 连击
    streakCounter.OnRoomRevealed(hasFlesh);

    // 第一个房开后出现[见好就收]
    if(openedRoomCount == 1 && !sellMidwayGroup.activeSelf) {
        sellMidwayGroup.SetActive(true);
        sellMidwayGroup.GetComponent<CanvasGroup>().DOFade(1f, 0.2f);
    }

    // 5房全开
    if(openedRoomCount >= 5) {
        allOpened = true;
        sellMidwayGroup.SetActive(false);
        OnAllRoomsOpened();
    }
}

private void UpdateEstimate() {
    int fullCount = 0;
    for(int i=0; i<openedRoomCount; i++)
        if(roomGrades[i] >= YieldGrade.High) fullCount++;

    float ratio = (float)fullCount / Mathf.Max(1, openedRoomCount);
    int remaining = 5 - openedRoomCount;
    float uncertainty = remaining / 5f * Random.Range(-0.3f, 0.3f);
    float estFull = fullCount + remaining * ratio * (1f + uncertainty);
    int estimate = Mathf.RoundToInt(estFull * 50f * (1f + shopManager.GetSellBonus()));
    sellMidwayPriceText.text = $"{estimate} 金币";
    sellMidwayPriceText.color = estimate >= lastEstimate ? Color.green : Color.red;
    lastEstimate = estimate;
}

public void OnSellMidway() {
    hasSoldMidway = true; sellMidwayGroup.SetActive(false);
    // 快速打开剩余房
    for(int i=openedRoomCount; i<5; i++) roomSlots[i].TryOpen();
    EventBus.Publish(new DurianMidwaySoldEvent(currentDurian, lastEstimate));
}

private async void OnAllRoomsOpened() {
    await UniTask.Delay(300);
    PlayerProgression.Instance.UnlockCollection(currentDurian.variety, EstimateOverallGrade(), lastEstimate);
    PlayerProgression.Instance.TotalOpens++; PlayerProgression.Instance.Save();
    ShowRating(EstimateOverallGrade());
    if(EstimateOverallGrade()==YieldGrade.Perfect || streakCounter.CurrentStreak>=3)
        ShowShareButton();
}

private YieldGrade EstimateOverallGrade() {
    int full=0; for(int i=0;i<5;i++) if(roomGrades[i]>=YieldGrade.High)full++;
    return full switch { 0=>YieldGrade.Empty, 1=>YieldGrade.Low, 2=>YieldGrade.Normal, 3=>YieldGrade.High, _=>YieldGrade.Perfect };
}

private async void ShowRating(YieldGrade g) { /* 评级弹出动画 */ }
private void ShowShareButton() { /* 微信分享 */ }
private void ClearRoomSlots() { foreach(var s in roomSlots) Destroy(s.gameObject); roomSlots.Clear(); }
```
