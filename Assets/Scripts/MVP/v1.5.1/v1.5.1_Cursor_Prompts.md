## S2b: 创建 ScratchCardRoom.cs（Sprite 裁剪版）

```
创建 Assets/_Project/Scripts/Component/ScratchCardRoom.cs：

using UnityEngine;
using UnityEngine.UI;
using UnityEngine.EventSystems;
using System;
using System.Collections.Generic;

public class ScratchCardRoom : MonoBehaviour, IPointerDownHandler, IDragHandler, IPointerUpHandler
{
    [Header("网格容器")]
    [SerializeField] private Transform gridParent;
    [SerializeField] private GameObject cellPrefab;  // 单个格子 Image

    [Header("配置引用")]
    [SerializeField] private ScratchCardConfig config;

    [Header("基础参数")]
    [SerializeField] private float revealThreshold = 0.3f;

    // 运行时数据
    private int gridSize;
    private List<Image> cellImages = new();
    private bool[] cellScratched;       // 每个格子是否已刮开
    private int totalCells;
    private int scratchedCount = 0;
    private bool isOpened = false;

    public int RoomIndex { get; set; }
    public YieldGrade RoomGrade { get; private set; }
    public float ScratchedRatio => (float)scratchedCount / totalCells;

    // 工具参数（由 ToolManager 设置）
    public float brushGridRadius = 1.5f;  // 手指覆盖约1.5格半径 → ~4格

    public Action<ScratchCardRoom> OnScratchUpdated;

    // ===== 初始化：裁剪 RC 和 RF 贴图 =====
    public void Init(int index, Sprite cover, Sprite flesh, YieldGrade grade)
    {
        RoomIndex = index;
        RoomGrade = grade;
        isOpened = false;
        gridSize = config.gridSize;
        totalCells = gridSize * gridSize;
        cellScratched = new bool[totalCells];
        scratchedCount = 0;

        // 确保贴图可读
        Texture2D rcTex = cover.texture;
        Texture2D rfTex = flesh.texture;
        float cellW = rcTex.width / (float)gridSize;
        float cellH = rcTex.height / (float)gridSize;

        // 清空旧格子
        foreach (var c in cellImages) Destroy(c.gameObject);
        cellImages.Clear();

        // 批量创建 gridSize×gridSize 个格子
        for (int y = 0; y < gridSize; y++)
        {
            for (int x = 0; x < gridSize; x++)
            {
                var cell = Instantiate(cellPrefab, gridParent).GetComponent<Image>();

                // 裁剪 RC 纹理的对应局部
                Rect rcRect = new Rect(x * cellW, y * cellH, cellW, cellH);
                Sprite rcSprite = Sprite.Create(rcTex, rcRect,
                    new Vector2(0.5f, 0.5f), 128f / gridSize);
                cell.sprite = rcSprite;

                cell.rectTransform.sizeDelta = new Vector2(128f/gridSize, 128f/gridSize);
                cell.rectTransform.anchoredPosition = new Vector2(
                    (x - gridSize/2f + 0.5f) * 128f/gridSize,
                    -(y - gridSize/2f + 0.5f) * 128f/gridSize);

                cellImages.Add(cell);
            }
        }

        // 预生成 RF 对应的格子 Sprite（不立即显示，切换时用）
        // 可以通过一个数组存储，这里简化为切换时动态裁剪
    }

    // ===== 刮卡交互 =====
    public void OnPointerDown(PointerEventData e) { Scratch(e); }
    public void OnDrag(PointerEventData e) { Scratch(e); }
    public void OnPointerUp(PointerEventData e)
    {
        if (ScratchedRatio >= revealThreshold)
            OnScratchUpdated?.Invoke(this);
    }

    private void Scratch(PointerEventData e)
    {
        if (isOpened) return;

        // 获取触摸点在 gridParent 中的本地坐标
        RectTransformUtility.ScreenPointToLocalPointInRectangle(
            gridParent as RectTransform, e.position, e.pressEventCamera, out Vector2 localPos);

        float halfSize = 64f; // 128/2
        float cellSize = 128f / gridSize;
        int cx = Mathf.Clamp((int)((localPos.x + halfSize) / cellSize), 0, gridSize - 1);
        int cy = Mathf.Clamp((int)((-localPos.y + halfSize) / cellSize), 0, gridSize - 1);

        int r = Mathf.CeilToInt(brushGridRadius);
        bool anyNew = false;

        for (int dy = -r; dy <= r; dy++)
        {
            for (int dx = -r; dx <= r; dx++)
            {
                if (dx * dx + dy * dy > brushGridRadius * brushGridRadius) continue;
                int gx = cx + dx, gy = cy + dy;
                if (gx < 0 || gx >= gridSize || gy < 0 || gy >= gridSize) continue;
                int idx = gy * gridSize + gx;
                if (!cellScratched[idx])
                {
                    cellScratched[idx] = true;
                    scratchedCount++;
                    // 切换为 RF 的对应局部
                    // 真实项目中这里换成预设的 rfSprites[idx]
                    cellImages[idx].color = new Color(0, 0, 0, 0); // 简化：透明=已刮
                    anyNew = true;
                }
            }
        }

        if (anyNew) OnScratchUpdated?.Invoke(this);
    }

    // ===== 一键全开 =====
    public void RevealAll()
    {
        isOpened = true;
        foreach (var cell in cellImages) cell.color = new Color(0, 0, 0, 0);
        scratchedCount = totalCells;
        OnScratchUpdated?.Invoke(this);
    }
}
```

**关键说明**：`cell.color = new Color(0,0,0,0)` 是简化写法。完整实现需在 `Init` 中预处理两组 Sprite 数组（rcSprites 和 rfSprites），刮开时切换 sprite。

在 Unity 中创建 `cellPrefab`：空 GameObject + Image 组件，尺寸自适应。

public class ToolManager : MonoBehaviour
{
    [Inject] private DurianSpriteConfig spriteConfig;
    [SerializeField] private ToolConfig config;

    public ToolDefinition CurrentTool { get; private set; }
    public event Action<ToolDefinition> OnToolChanged;

    private void Start()
    {
        CurrentTool = config.tools[0]; // 默认手指
    }

    public void SelectTool(int index)
    {
        var tool = config.tools[index];
        if (tool.costGold > 0 && PlayerData.Instance.Gold < tool.costGold) return;
        if (tool.costGold == -1) { /* 触发广告 */ return; }

        if (tool.costGold > 0) PlayerData.Instance.Gold -= tool.costGold;
        CurrentTool = tool;
        OnToolChanged?.Invoke(tool);
    }
}

// 不再实例化 RoomSlot，改为实例化 ScratchCardRoom Prefab（挂载了 ScratchCardRoom 组件）

private List<ScratchCardRoom> rooms = new();

public void PrepareOpen(DurianData durian)
{
    // ... 前置逻辑不变 ...

    ClearRooms();
    var covers = spriteConfig.GetRcJz();
    for (int i = 0; i < 5; i++)
    {
        var go = Instantiate(scratchRoomPrefab, roomSlotsParent);
        go.GetComponent<RectTransform>().anchoredPosition = roomPositions[i];
        var room = go.GetComponent<ScratchCardRoom>();
        room.Init(i, covers[i], spriteConfig.GetRf(roomGrades[i]), roomGrades[i]);
        room.OnScratchUpdated += OnRoomScratched;
        rooms.Add(room);
    }
    // ...
}

private void OnRoomScratched(ScratchCardRoom room)
{
    if (hasSoldMidway || allOpened) return;

    float totalScratched = 0;
    float weightedFlesh = 0;
    int anyScratched = 0;
    foreach (var r in rooms)
    {
        if (r.ScratchedRatio > 0.1f)
        {
            totalScratched += r.ScratchedRatio;
            weightedFlesh += r.ScratchedRatio * r.GetFleshDensity();
            anyScratched++;
        }
    }

    // 更新估值
    UpdateEstimate(avgScratched, avgFlesh);
    
    // 任意房刮开30%以上→出现卖出按钮
    if (anyScratched > 0 && avgScratched >= 0.3f && !sellMidwayGroup.activeSelf)
    {
        sellMidwayGroup.SetActive(true);
        sellMidwayCanvas.DOFade(1f, 0.2f);
    }
}

private int lastEstimate = 0;
private void UpdateEstimate(float avgScratched, float avgFlesh)
{
    int estimate = Mathf.RoundToInt(avgScratched * avgFlesh * 250f * (1 + shopManager.GetSellBonus()));

    // 超过上次值→金色大涨，低于→红色暴跌
    bool isUp = estimate > lastEstimate;
    bool isBig = Mathf.Abs(estimate - lastEstimate) > 50;

    // 🔥 核心：剧烈跳动的数字
    sellMidwayPriceText.text = estimate + " 金币";
    
    // 颜色：涨=金色(#FFD700)，跌=红色(#FF4444)，幅度越大颜色越深
    sellMidwayPriceText.color = isUp 
        ? (isBig ? new Color(1f, 0.7f, 0f) : new Color(1f, 0.85f, 0f)) 
        : (isBig ? new Color(1f, 0.2f, 0.1f) : new Color(1f, 0.4f, 0.2f));
    
    // 缩放跳动：大幅变化时弹得更高
    float punchScale = isBig ? 0.35f : 0.15f;
    sellMidwayPriceText.transform.DOPunchScale(Vector3.one * punchScale, 0.25f, 3, 0.5f);
    
    // 🔥 大幅变化时额外特效：文字外发光模拟
    if (isBig)
    {
        var outline = sellMidwayPriceText.GetComponent<Outline>();
        if (outline != null)
        {
            outline.effectColor = isUp ? new Color(1f, 0.8f, 0f, 0.6f) : new Color(1f, 0f, 0f, 0.6f);
            outline.effectDistance = new Vector2(3, 3);
            DOTween.To(() => outline.effectDistance, 
                v => outline.effectDistance = v,
                new Vector2(0, 0), 0.5f).SetEase(Ease.OutQuad);
        }
    }
    
    lastEstimate = estimate;
}
```

## S8b: 创建 CrackLineOverlay.cs（壳盖裂开缝装饰线）

```
创建 Assets/_Project/Scripts/Component/CrackLineOverlay.cs：

壳盖裂开缝动画——在 ScratchCardRoom 的壳盖上方叠加裂纹线，
当玩家开始刮时，裂纹从中间向两侧扩散，最后壳盖"翻起"。
解决"纯格子刮卡不像开榴莲"的问题。

using UnityEngine;
using UnityEngine.UI;
using DG.Tweening;

public class CrackLineOverlay : MonoBehaviour
{
    [SerializeField] private Image crackLineImage;  // 裂开缝贴图（一张细长的裂纹线）
    [SerializeField] private Image leftShellFlip;    // 左半壳翻开部分
    [SerializeField] private Image rightShellFlip;   // 右半壳翻开部分

    private float crackProgress = 0f;  // 0=完整, 1=完全裂开

    public void ShowCrack(float progress)
    {
        // progress: 0-1, 从刮卡进度映射
        crackProgress = Mathf.Clamp01(progress);

        // 裂纹线扩大：从中间一条细缝 → 向两侧展开
        crackLineImage.rectTransform.sizeDelta = new Vector2(
            2 + crackProgress * 60f,  // 缝宽从2px到62px
            crackLineImage.rectTransform.sizeDelta.y);

        // 透明度：progress>20%时裂纹可见
        float alpha = Mathf.Clamp01((crackProgress - 0.15f) / 0.15f);
        crackLineImage.color = new Color(0.2f, 0.15f, 0.05f, alpha);

        // 壳盖翻起：progress>50%时左右壳开始微翻
        if (crackProgress > 0.5f)
        {
            float flipAngle = (crackProgress - 0.5f) * 2f * 20f;  // 最大翻起20度
            leftShellFlip.transform.localRotation = Quaternion.Euler(0, 0, -flipAngle);
            rightShellFlip.transform.localRotation = Quaternion.Euler(0, 0, flipAngle);
            leftShellFlip.color = new Color(1, 1, 1, (crackProgress - 0.5f) * 2f);
            rightShellFlip.color = new Color(1, 1, 1, (crackProgress - 0.5f) * 2f);
        }
    }

    public void Reset()
    {
        crackProgress = 0f;
        crackLineImage.color = Color.clear;
        leftShellFlip.color = Color.clear;
        rightShellFlip.color = Color.clear;
        leftShellFlip.transform.localRotation = Quaternion.identity;
        rightShellFlip.transform.localRotation = Quaternion.identity;
    }
}
```

# Step S10: 🟡 连击火焰在开果过程实时显示（竞品优化 P1）

```
修改 StreakCounter.cs，火焰不在结算时才显示，而是开果过程中实时叠加在榴莲上方：

public class StreakCounter : MonoBehaviour
{
    [SerializeField] private Image flameOverlay;  // CO-01 火焰，叠在榴莲图上方
    [SerializeField] private CanvasGroup flameCanvas;

    private int streak = 0;

    // 每开一个满肉房→火焰升一级
    public void OnRoomRevealed(YieldGrade grade)
    {
        if (grade >= YieldGrade.Good)
        {
            streak++;
            UpdateFlame(streak);
        }
        else
        {
            // 空房→火焰熄灭（保留之前的等级但变暗）
            flameCanvas.DOFade(0.2f, 0.3f);
        }
    }

    private void UpdateFlame(int level)
    {
        // 火焰透明度和缩放随连击等级变化
        float alpha = Mathf.Lerp(0.3f, 1f, level / 5f);
        float scale = Mathf.Lerp(0.8f, 1.3f, level / 5f);

        flameCanvas.DOFade(alpha, 0.2f);
        flameOverlay.transform.DOScale(scale, 0.2f).SetEase(Ease.OutBack);
        flameOverlay.transform.DOShakeRotation(0.3f, 5f);  // 火焰抖动

        // 连击数字在火焰中央
        comboText.text = streak >= 2 ? $"×{streak}" : "";
    }

    public void Reset()
    {
        streak = 0;
        flameCanvas.DOFade(0f, 0.2f);
        comboText.text = "";
    }
}

public class CustomerBidSystem : MonoBehaviour
{
    [SerializeField] private GameObject bidPopup;        // 出价弹窗根节点
    [SerializeField] private Image customerAvatar;       // 客户头像
    [SerializeField] private Text customerNameText;      // "富豪赵总"

    private bool bidActive = false;

    [Inject] private DurianSpriteConfig spriteConfig;

    void Start()
    {
        customers = new CustomerProfile[]
        {
            new() { name="白领小李",  personality="保守", budgetRange="100-300",
                    avatar=spriteConfig.npcOfficeWorker },
            new() { name="富豪赵总",  personality="冲动", budgetRange="300-800",
                    avatar=spriteConfig.npcRichBoss },
            new() { name="美食博主",  personality="跟风", budgetRange="200-600",
                    avatar=spriteConfig.npcFoodBlogger },
            new() { name="水果摊同行", personality="精明", budgetRange="50-200",
                    avatar=spriteConfig.npcFruitVendor },
            new() { name="神秘买家",  personality="随机", budgetRange="500-1500",
                    avatar=spriteConfig.npcMysteryBuyer },
        };

        acceptBtn.onClick.AddListener(AcceptBid);
        rejectBtn.onClick.AddListener(RejectBid);
        bidPopup.SetActive(false);
    }

    /// <summary>尝试触发客户出价</summary>
    public bool TryTriggerBid(float scratchedRatio, int currentEstimate,
                               System.Action<int> onAccept)
    {
        if (bidActive) return false;                       // 已有出价进行中
        if (scratchedRatio < 0.25f) return false;          // 刮太少不出价
        if (Random.value > 0.45f) return false;            // 45%几率触发

        var customer = customers[Random.Range(0, customers.Length)];
        onAcceptBid = onAccept;

        // 计算出价（基于当前估价+客户性格波动）
        float personalityFactor = customer.personality switch
        {
            "冲动" => Random.Range(1.1f, 1.5f),   // 冲动型出高价
            "保守" => Random.Range(0.8f, 1.05f),  // 保守型谨慎出价
            "精明" => Random.Range(0.7f, 0.95f),  // 精明型压价
            "跟风" => Random.Range(0.9f, 1.2f),   // 跟风型中等
            "随机" => Random.Range(0.5f, 2.0f),   // 神秘买家大起大落
            _ => 1.0f
        };
        currentBid = Mathf.RoundToInt(currentEstimate * personalityFactor);

        ShowBidPopup(customer, currentBid);
        return true;
    }

    private void ShowBidPopup(CustomerProfile customer, int bid)
    {
        bidActive = true;
        customerAvatar.sprite = customer.avatar;
        customerNameText.text = $"{customer.name} 看中了你的榴莲！";
        bidAmountText.text = $"出价: {bid} 金币";
        countdown = 15f;
        UpdateCountdownDisplay();

        // 弹窗从底部弹入
        bidPopup.SetActive(true);
        popupCanvas.alpha = 0;
        bidPopup.transform.localScale = Vector3.one * 0.8f;
        popupCanvas.DOFade(1f, 0.3f);
        bidPopup.transform.DOScale(1f, 0.3f).SetEase(Ease.OutBack);
    }

    void Update()
    {
        if (!bidActive) return;
        countdown -= Time.deltaTime;
        UpdateCountdownDisplay();

        if (countdown <= 0)
        {
            // 超时 → 客户走了，自动拒绝
            RejectBid();
        }

        // 最后5秒红色闪烁
        if (countdown <= 5f)
        {
            countdownText.color = Color.Lerp(Color.red, Color.yellow,
                Mathf.PingPong(Time.time * 3f, 1f));
        }
    }

    void UpdateCountdownDisplay()
    {
        countdownText.text = $"剩余 {Mathf.CeilToInt(countdown):D2}:{Mathf.FloorToInt((countdown % 1) * 100):D2}";
    }

    void AcceptBid()
    {
        bidActive = false;
        bidPopup.SetActive(false);
        onAcceptBid?.Invoke(currentBid);
    }

    void RejectBid()
    {
        bidActive = false;
        bidPopup.SetActive(false);  // 客户离开，同一个榴莲不会再出价
    }
}

```

// 每天从4种订单模板中刷新1个，3天后过期
// 完成订单→大额金币奖励（比正常卖高50%）

[System.Serializable]
public class CustomerOrder
{

    public string orderName;          // "富太太的宴会用果"

}

public class CustomerOrderSystem : MonoBehaviour
{
    private CustomerOrder[] orderTemplates = new[]
    {
        new() { orderName="富太太的宴会用果", customerName="富太太",
                variety=DurianVariety.Jinzheng, minGrade=YieldGrade.Burst,
                baseReward=600 },
        new() { orderName="美食博主的挑战", customerName="美食博主",
                variety=DurianVariety.MusangKing, minGrade=YieldGrade.Good,
                baseReward=400 },
        new() { orderName="水果店的补货单", customerName="水果摊同行",
                variety=DurianVariety.Ganyao, minGrade=YieldGrade.Normal,

        new() { orderName="神秘买家的收藏", customerName="神秘买家",
                variety=DurianVariety.Any, minGrade=YieldGrade.Burst,
                baseReward=1000 },
    };

    private CustomerOrder currentOrder;

    public void RefreshDailyOrder()
    {
        // 每天随机刷1个新订单（替换过期/已完成的）
        if (currentOrder == null || currentOrder.daysRemaining <= 0 || currentOrder.completed)
        {
            currentOrder = Instantiate(orderTemplates[Random.Range(0, orderTemplates.Length)]);
            currentOrder.orderId = Guid.NewGuid().ToString();
            currentOrder.daysRemaining = 3;
        }
    }

    public void CheckOrderCompletion(DurianData durian, YieldGrade result)
    {
        if (currentOrder == null || currentOrder.completed) return;

        if ((currentOrder.variety == DurianVariety.Any ||
             durian.variety == currentOrder.variety) &&
            result >= currentOrder.minGrade)
        {
            CompleteOrder();
        }
    }

    private void CompleteOrder()
