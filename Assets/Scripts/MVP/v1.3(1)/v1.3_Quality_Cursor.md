```csharp
public class PlayerProgression
{
    public static PlayerProgression Instance;

    public int[] CollectionBestPrice = new int[18];     // 每格最佳售价
    
    // ===== 成就 =====

    public bool[] AchievementClaimed = new bool[12];    // 奖励已领取

    public int[] DailyQuestTargets = new int[3];        // 今日任务目标
    public string LastDailyResetDate;                   // 上次重置日期
    
    /// <summary>计算图鉴解锁格对应的索引</summary>
    public static int GetCollectionIndex(VarietyType variety, YieldGrade grade)
    {
        return (int)variety * 6 + (int)grade;
    }
    
    /// <summary>解锁图鉴格</summary>
    public void UnlockCollection(VarietyType variety, YieldGrade grade, int sellPrice)
    {
        int idx = GetCollectionIndex(variety, grade);
        CollectionUnlocked[idx] = true;
        if (sellPrice > CollectionBestPrice[idx])
            CollectionBestPrice[idx] = sellPrice;
        
        Save();
    }
    
    /// <summary>获取图鉴完成度（0~18）</summary>
    public int GetCollectionCount()
    {
        int count = 0;
        for (int i = 0; i < 18; i++)
            if (CollectionUnlocked[i]) count++;
        return count;
    }
    
    /// <summary>增加经验</summary>
    public void AddExp(int amount)
    {
        Experience += amount;
        while (Experience >= ExpToNextLevel && Level < MaxLevel)
        {
            Experience -= ExpToNextLevel;
            Level++;
            EventBus.Publish(new LevelUpEvent(Level));
        }
        Save();
    }
    
    public void Save()
    {
        PlayerPrefs.SetInt("Level", Level);
        PlayerPrefs.SetInt("Experience", Experience);
        PlayerPrefs.SetInt("TotalOpens", TotalOpens);
        // ... 其余持久化字段
        PlayerPrefs.Save();
    }
    
    public void Load()
    {
        Level = PlayerPrefs.GetInt("Level", 1);
        Experience = PlayerPrefs.GetInt("Experience", 0);
        // ... 其余加载
    }
}
```

```csharp
public class CollectionPage : MonoBehaviour
{
    [Inject] private DurianSpriteConfig spriteConfig;
    
    [SerializeField] private Transform gridParent;       // 网格容器
    [SerializeField] private GameObject collectionSlotPrefab; // 单格预设
    [SerializeField] private Image progressBar;          // 顶部进度条

    private CollectionSlot[] slots = new CollectionSlot[18];
    
    private void Start()
    {
        CreateGrid();
        UpdateDisplay();
    }
    
    private void CreateGrid()
    {
        string[] varietyNames = { "金枕", "干尧", "猫山王" };
        string[] gradeNames = { "空壳", "少肉", "正常", "满肉", "爆肉", "榴莲之王" };
        
        for (int v = 0; v < 3; v++)
        {
            for (int g = 0; g < 6; g++)
            {
                var slotObj = Instantiate(collectionSlotPrefab, gridParent);
                var slot = slotObj.GetComponent<CollectionSlot>();
                
                int idx = PlayerProgression.GetCollectionIndex((VarietyType)v, (YieldGrade)g);
                bool unlocked = PlayerProgression.Instance.CollectionUnlocked[idx];
                
                slot.Initialize(
                    varietyNames[v],
                    gradeNames[g],
                    unlocked,
                    unlocked ? PlayerProgression.Instance.CollectionBestPrice[idx] : 0,
                    spriteConfig  // 用于获取对应评级图标
                );
                
                slots[idx] = slot;
            }
        }
    }
    
    private void UpdateDisplay()
    {
        int collected = PlayerProgression.Instance.GetCollectionCount();
        progressBar.fillAmount = collected / 18f;
        progressText.text = $"{collected}/18";
        
        // 检查行完成奖励
        for (int v = 0; v < 3; v++)
        {
            bool rowComplete = true;
            for (int g = 0; g < 6; g++)
            {
                int idx = v * 6 + g;
                if (!PlayerProgression.Instance.CollectionUnlocked[idx])
                    rowComplete = false;
            }
            if (rowComplete)
            {
                slots[v * 6].ShowRowCompleteBadge(); // 行首显示完成徽章
            }
        }
    }
}
```

```csharp
public class ComboSystem
{
    public int CurrentCombo { get; private set; } = 0;
    public bool HasComboProtection { get; set; } = false; // 连击保护道具
    
    /// <summary>开果结算后调用</summary>
    public void OnDurianOpened(YieldGrade grade)
    {
        if (grade >= YieldGrade.Normal)
        {
            CurrentCombo++;
            if (CurrentCombo > PlayerProgression.Instance.BestCombo)
            {
                PlayerProgression.Instance.BestCombo = CurrentCombo;
            }
        }
        else
        {
            if (HasComboProtection)
            {
                HasComboProtection = false;
                // 不掉连击
                EventBus.Publish(new ComboProtectedEvent());
            }
            else
            {
                CurrentCombo = 0;
                EventBus.Publish(new ComboBrokenEvent());
            }
        }
        
        EventBus.Publish(new ComboUpdatedEvent(CurrentCombo));
    }
    
    /// <summary>获取连击售卖加成系数</summary>
    public float GetComboMultiplier()
    {
        return CurrentCombo switch
        {
            < 3 => 1.0f,
            < 5 => 1.2f,
            < 7 => 1.5f,
            < 10 => 2.0f,
            _ => 3.0f
        };
    }
    
    /// <summary>获取连击等级（用于视觉特效）</summary>
    public int GetComboLevel()
    {
        return CurrentCombo switch
        {
            < 3 => 0,
            < 5 => 1,
            < 7 => 2,
            < 10 => 3,
            _ => 4
        };
    }
    
    /// <summary>重置连击（用于测试/强制重置）</summary>
    public void ResetCombo()
    {
        CurrentCombo = 0;
        EventBus.Publish(new ComboUpdatedEvent(0));
    }
}
```

```csharp
public class CriticalLuckSystem
{
    private const float BASE_CRIT_RATE = 0.08f; // 基础暴击率8%
    
    /// <summary>开果时判断是否触发暴击</summary>
    /// <returns>(是否暴击, 暴击后的出肉率修正值, 暴击类型)</returns>
    public (bool isCrit, float yieldBonus, string critType) RollCritical(
        VarietyType variety, AppearanceType appearance, YieldGrade originalGrade)
    {
        float critRate = BASE_CRIT_RATE;
        
        // 外观影响暴击率
        critRate *= appearance switch
        {
            AppearanceType.Poor => 0.25f,     // 劣质: 2%
            AppearanceType.Normal => 1.0f,     // 普通: 8%
            AppearanceType.Good => 1.5f,        // 优质: 12%
            AppearanceType.Premium => 2.5f,     // 极品: 20%
            _ => 1.0f
        };
        
        // 幸运符道具翻倍
        if (HasLuckyCharm)
        {
            critRate *= 2f;
            HasLuckyCharm = false;
        }
        
        if (Random.value < critRate)
        {
            // 暴击触发！
            float bonus;
            string critType;
            
            if (appearance == AppearanceType.Poor && Random.value < 0.25f)
            {
                // 劣质外表的"绝地翻盘"：直接跳爆肉
                bonus = 0.5f;
                critType = "绝地翻盘";
            }
            else
            {
                bonus = 0.1f; // 出肉率+10%
                critType = "幸运暴击";
            }
            
            PlayerProgression.Instance.CriticalHits++;
            EventBus.Publish(new CriticalHitEvent(critType, bonus));
            return (true, bonus, critType);
        }
        
        return (false, 0f, null);
    }
    
    public bool HasLuckyCharm { get; set; } = false;
}
```

```csharp
        new() { id=6, description="看1次广告", target=1, reward=30, type=QuestType.WatchAd },
        new() { id=7, description="解锁1个新图鉴格", target=1, reward=150, type=QuestType.NewCollection },
        new() { id=8, description="达成3连击", target=3, reward=100, type=QuestType.ReachCombo },
    };
    
    /// <summary>每日刷新3个随机任务</summary>
    public void RefreshDailyQuests()
    {
        var shuffled = questPool.OrderBy(x => Random.value).Take(3).ToArray();
        var p = PlayerProgression.Instance;
        
        for (int i = 0; i < 3; i++)
        {
            p.DailyQuestIds[i] = shuffled[i].id;
            p.DailyQuestProgress[i] = 0;
            p.DailyQuestTargets[i] = shuffled[i].target;
        }
        
        p.LastDailyResetDate = DateTime.Now.ToString("yyyy-MM-dd");
        p.Save();
    }
    
    /// <summary>检查是否需要跨天重置</summary>
    public void CheckDailyReset()
    {
        var p = PlayerProgression.Instance;
        string today = DateTime.Now.ToString("yyyy-MM-dd");
        if (p.LastDailyResetDate != today)
        {
            RefreshDailyQuests();
        }
    }
    
    /// <summary>更新任务进度</summary>
    public void UpdateProgress(QuestType type, int amount = 1)
    {
        var p = PlayerProgression.Instance;
        for (int i = 0; i < 3; i++)
        {
            var quest = questPool.FirstOrDefault(q => q.id == p.DailyQuestIds[i]);
            if (quest.type == type)
            {
                p.DailyQuestProgress[i] = Mathf.Min(
                    p.DailyQuestProgress[i] + amount, p.DailyQuestTargets[i]);
            }
        }
    }
    
    /// <summary>领取任务奖励</summary>
    public bool ClaimQuestReward(int slotIndex)
    {
        var p = PlayerProgression.Instance;
        if (p.DailyQuestProgress[slotIndex] < p.DailyQuestTargets[slotIndex])
            return false;
        
        var quest = questPool.First(q => q.id == p.DailyQuestIds[slotIndex]);
        PlayerData.Instance.Gold += quest.reward;
        
        // 标记为已领取（进度设为-1）
        p.DailyQuestProgress[slotIndex] = -1;
        p.Save();
        
        return true;
    }
}
```

```csharp
        }
```
