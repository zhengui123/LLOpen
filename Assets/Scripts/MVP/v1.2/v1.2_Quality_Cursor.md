```
## Step Q1.2: 重写 DurianOpener —— 开裂序列动画

### Cursor 提示词
```
请完全重写 `Assets/_Project/Scripts/Component/DurianOpener.cs` 中的开果逻辑。
旧逻辑（v1.0 左右壳分离）全部替换为新的开裂叠加方案。

**新设计**：
- 榴莲完整图始终在画面中央不动
- 开裂叠加图（CP-01~05）在榴莲上方逐级切换
- 开裂进度由 KnifeTool 的滑动距离驱动

```csharp
using System;
using UnityEngine;
using UnityEngine.UI;
using DG.Tweening;
using Cysharp.Threading.Tasks;
using VContainer;

public class DurianOpener : MonoBehaviour
{
    [Inject] private DurianSpriteConfig spriteConfig;
    
    [Header("画面中央元素")]
    [SerializeField] private Image wholeDurianImage;    // 榴莲完整图（始终显示）
    [SerializeField] private Image crackOverlayImage;   // 开裂叠加图（叠在榴莲上方）
    [SerializeField] private Image openedDurianImage;   // 已开榴莲图（最终显示）
    
    [Header("评级UI")]
    [SerializeField] private Image ratingBadgeImage;    // 评级图标
    [SerializeField] private CanvasGroup ratingCanvasGroup;
    
    [Header("果肉揭示区")]
    [SerializeField] private Transform fleshGridParent; // 果肉/空房块的父节点
    [SerializeField] private GameObject fleshRoomPrefab; // 单房预设（含 Image）
    
    private float currentCrackProgress = 0f;
    
    private void Start()
    {
        crackOverlayImage.enabled = false;
        openedDurianImage.enabled = false;
        ratingCanvasGroup.alpha = 0f;
    }
    
    /// <summary>KnifeTool 每帧调用，更新开裂进度</summary>
    public void UpdateCrackProgress(float progress)
    {
        progress = Mathf.Clamp01(progress);
        currentCrackProgress = progress;
        
        if (!crackOverlayImage.enabled && progress > 0f)
        {
            crackOverlayImage.enabled = true;
        }
        
        // 根据进度切换开裂阶段贴图
        crackOverlayImage.sprite = spriteConfig.GetCrackStage(progress);
    }
    
    /// <summary>滑动完成，开始揭示果肉</summary>
    public async UniTask OnSwipeComplete(DurianData durian)
    {
        // Step 1: 开裂到最终阶段（如果还没到）
        crackOverlayImage.sprite = spriteConfig.crackStage5;
        await UniTask.Delay(300);
        
        // Step 2: 开裂叠加图淡出，已开榴莲图淡入
        crackOverlayImage.DOFade(0f, 0.3f);
        openedDurianImage.sprite = spriteConfig.GetOpenedSprite(
            durian.variety, durian.yieldGrade);
        openedDurianImage.enabled = true;
        openedDurianImage.DOFade(1f, 0.3f);
        
        await UniTask.Delay(350);
        
        // Step 3: 逐房揭示果肉
        await RevealFleshRooms(durian.roomCount, durian.roomResults);
        
        // Step 4: 显示评级
        await ShowRating(durian.yieldGrade);
    }
    
    private async UniTask RevealFleshRooms(int roomCount, bool[] roomResults)
    {
        for (int i = 0; i < roomCount; i++)
        {
            var roomObj = Instantiate(fleshRoomPrefab, fleshGridParent);
            roomObj.transform.localPosition = GetRoomPosition(i, roomCount);
            
            var renderer = roomObj.GetComponent<Image>();
            renderer.sprite = roomResults[i]
                ? spriteConfig.fleshPiece   // FL-01
                : spriteConfig.emptyPiece;  // FL-02
            
            // DOTween 弹出动画
            roomObj.transform.localScale = Vector3.zero;
            roomObj.transform.DOScale(1f, 0.3f).SetEase(Ease.OutBack);
            
            await UniTask.Delay(150);
        }
    }
    
    private async UniTask ShowRating(YieldGrade grade)
    {
        ratingBadgeImage.sprite = spriteConfig.GetRatingIcon(grade);
        
        // 弹出 + 放大动画
        ratingCanvasGroup.DOFade(1f, 0.3f);
        ratingBadgeImage.transform.DOPunchScale(
            new Vector3(0.3f, 0.3f, 0f), 0.5f, 2, 0.5f);
        
        // 榴莲之王特殊处理
        if (grade == YieldGrade.Perfect)
        {
            ratingBadgeImage.sprite = spriteConfig.kingRating;
            // 额外彩虹光晕
        }
        
        await UniTask.Delay(1500);
    }
    
    private Vector3 GetRoomPosition(int index, int total)
    {
        float startX = -(total - 1) * 40f;
        return new Vector3(startX + index * 80f, 0f, 0f);
    }
}
```

**关键改动**：
- `wholeDurianImage` 始终在中间，不移动
- `crackOverlayImage` 是叠加层，放在 wholeDurianImage 的**上方**（Canvas Sort Order +1）
- `openedDurianImage` 在开裂完成后替换显示
- 所有动画使用 DOTween，不直接操作 Transform
```

## Step Q1.3: 修改 KnifeTool —— 进度映射

### Cursor 提示词
```
请修改 `Assets/_Project/Scripts/Component/KnifeTool.cs`，将滑动距离映射为开裂进度(0.0~1.0)：

```csharp
public class KnifeTool : MonoBehaviour
{
    [Inject] private DurianSpriteConfig spriteConfig;
    [Inject] private DurianOpener durianOpener;
    
    [SerializeField] private Image knifeImage;
    [SerializeField] private Image swipeGuideImage;
    [SerializeField] private float swipeThreshold = 200f; // 触发完成的滑动距离
    
    private bool isSwiping = false;
    private float swipeStartY;
    private float swipeDistance;
    private DurianData currentDurian;
    
    private void Start()
    {
        knifeImage.sprite = spriteConfig.knifeSprite; // K-01
        knifeImage.enabled = false;
        swipeGuideImage.sprite = spriteConfig.swipeGuideIcon; // UI-07
    }
    
    public void BeginSwipe(DurianData durian)
    {
        currentDurian = durian;
        isSwiping = true;
        swipeDistance = 0f;
        knifeImage.enabled = true;
        swipeGuideImage.enabled = false; // 开始滑动后隐藏引导
        
        // 刀出现在榴莲顶部中心
        knifeImage.rectTransform.anchoredPosition = new Vector2(0, 150f);
        wholeDurianImage.sprite = spriteConfig.GetUnopenedSprite(
            durian.variety, durian.appearance);
    }
    
    private void Update()
    {
        if (!isSwiping) return;
        
        if (Input.GetMouseButton(0) || Input.touchCount > 0)
        {
            Vector2 touchPos = Input.touchCount > 0 
                ? Input.GetTouch(0).position 
                : (Vector2)Input.mousePosition;
            
            float deltaY = touchPos.y - swipeStartY;
            swipeDistance = Mathf.Abs(deltaY);
            
            // 刀跟随手指垂直移动
            knifeImage.rectTransform.anchoredPosition = new Vector2(0, 150f - swipeDistance * 0.5f);
            
            // 🔑 关键：滑动距离映射为开裂进度 0.0~1.0
            float progress = Mathf.Clamp01(swipeDistance / swipeThreshold);
            durianOpener.UpdateCrackProgress(progress);
            
            if (swipeDistance >= swipeThreshold)
            {
                OnSwipeComplete();
            }
        }
    }
    
    private void OnSwipeComplete()
    {
        isSwiping = false;
        
        // 刀渐隐退出
        knifeImage.DOFade(0f, 0.3f).OnComplete(() => knifeImage.enabled = false);
        
        // 触发开果完成
        durianOpener.OnSwipeComplete(currentDurian).Forget();
    }
}
```

**关键改动**：
- 去掉旧的左右壳分离调用
- 改为 `durianOpener.UpdateCrackProgress(progress)` 逐帧更新开裂
- 刀贴图（K-01）不删，仍然需要，它在榴莲上方跟随手指移动
```

## Step Q2.5: 全局页面过渡动画

### Cursor 提示词
```
请创建 `Assets/_Project/Scripts/View/PageTransition.cs` 实现页面切换过渡：

```csharp
public class PageTransition : MonoBehaviour
{
    public enum Direction { FromRight, FromLeft }
    
    public static async UniTask TransitionTo(
        CanvasGroup fromPage, CanvasGroup toPage, Direction direction = Direction.FromRight)
    {
        float slideDistance = direction == Direction.FromRight ? 200f : -200f;
        
        // 新页面从侧边滑入
        toPage.alpha = 0f;
        toPage.transform.localPosition = new Vector3(slideDistance, 0, 0);
        toPage.gameObject.SetActive(true);
        
        await UniTask.WhenAll(
            toPage.transform.DOLocalMoveX(0f, 0.3f).SetEase(Ease.OutQuad).ToUniTask(),
            toPage.DOFade(1f, 0.3f).ToUniTask(),
            fromPage.transform.DOLocalMoveX(-slideDistance, 0.3f).SetEase(Ease.InQuad).ToUniTask(),
            fromPage.DOFade(0f, 0.3f).ToUniTask()
        );
        
        fromPage.gameObject.SetActive(false);
        fromPage.transform.localPosition = Vector3.zero;
        fromPage.alpha = 1f;
    }
}
```

在所有页面切换处（MarketPage→OpenPage→SellPage→MarketPage）调用此方法。
